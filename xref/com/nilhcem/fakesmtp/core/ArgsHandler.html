53  	/**
54  	 * Handles command line arguments.
55  	 */
56  	private ArgsHandler() {
57  		options = new Options();
58  		options.addOption(OPT_EMAILS_DIR_SHORT, OPT_EMAILS_DIR_LONG, true, OPT_EMAILS_DESC);
59  		options.addOption(OPT_AUTOSTART_SHORT, OPT_AUTOSTART_LONG, false, OPT_AUTOSTART_DESC);
60  		options.addOption(OPT_PORT_SHORT, OPT_PORT_LONG, true, OPT_PORT_DESC);
61  		options.addOption(OPT_BACKGROUNDSTART_SHORT, OPT_BACKGROUNDSTART_LONG, false, OPT_BACKGROUNDSTART_DESC);
62  		options.addOption(OPT_RELAYDOMAINS_SHORT, OPT_RELAYDOMAINS_LONG, true, OPT_RELAYDOMAINS_DESC);
63  	}

65  	/**
66  	 * Interprets command line arguments.
67  	 *
68  	 * @param args program's arguments.
69  	 * @throws ParseException when arguments are invalid.
70  	 */
71  	public void handleArgs(String[] args) throws ParseException {
72  		CommandLineParser parser = new GnuParser();
73  		CommandLine cmd = parser.parse(options, args);
74  
75  		String outputDir = cmd.getOptionValue(OPT_EMAILS_DIR_SHORT);
76  		if (outputDir != null) {
77  			UIModel.INSTANCE.setSavePath(outputDir);
78  		}
79  
80  		port = cmd.getOptionValue(OPT_PORT_SHORT);
81  		startServerAtLaunch = cmd.hasOption(OPT_AUTOSTART_SHORT);
82  		backgroundStart = cmd.hasOption(OPT_BACKGROUNDSTART_SHORT);
83  
84  		String relaydomains = cmd.getOptionValue(OPT_RELAYDOMAINS_SHORT);
85  		if (relaydomains != null) {
86  			List<String> domains = new ArrayList<String>();
87  			for (String domain : Arrays.asList(relaydomains.split(OPT_RELAYDOMAINS_SEPARATOR))) {
88  				domains.add(domain.trim());
89  			}
90  			UIModel.INSTANCE.setRelayDomains(domains);
91  		}
92  	}
