45  		options.addOption(OPT_EMAILS_DIR_SHORT, OPT_EMAILS_DIR_LONG, true, OPT_EMAILS_DESC);
46  		options.addOption(OPT_AUTOSTART_SHORT, OPT_AUTOSTART_LONG, false, OPT_AUTOSTART_DESC);
47  		options.addOption(OPT_PORT_SHORT, OPT_PORT_LONG, true, OPT_PORT_DESC);
48  	}

50  	/**
51  	 * Interprets command line arguments.
52  	 *
53  	 * @param args program's arguments.
54  	 * @throws ParseException when arguments are invalid.
55  	 */
56  	public void handleArgs(String[] args) throws ParseException {
57  		CommandLineParser parser = new GnuParser();
58  		CommandLine cmd = parser.parse(options, args);
59  
60  		String outputDir = cmd.getOptionValue(OPT_EMAILS_DIR_SHORT);
61  		if (outputDir != null) {
62  			UIModel.INSTANCE.setSavePath(outputDir);
63  		}
64  
65  		port = cmd.getOptionValue(OPT_PORT_SHORT);
66  		startServerAtLaunch = cmd.hasOption(OPT_AUTOSTART_SHORT);
67  	}
68  
69  	/**

73  	 */
74  	public void displayUsage(ParseException exception) {
75  		HelpFormatter formatter = new HelpFormatter();
76  		formatter.printHelp(String.format(Locale.US, "java -jar %s [OPTION]...", getJarName()), options);
77  	}
