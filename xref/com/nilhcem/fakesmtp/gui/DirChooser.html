13  /**
14   * Provides a graphical directory chooser dialog.
15   * <p>
16   * The directory chooser is used to select the folder where emails will be saved in.<br />
17   * It can be launched from the menu bar, or from the main panel.
18   * </p>
19   *
20   * @author Nilhcem
21   * @since 1.0
22   */
23  public final class DirChooser extends Observable implements Observer {
24  	private final JFileChooser dirChooser = new JFileChooser();
25  	private Component parent = null;
26  
27  	/**
28  	 * Creates a {@code JFileChooser} component and sets it to be for directories only.
29  	 *
30  	 * @param parent the component from where the chooser will be launched <i>(should be the main panel of the application)</i>.
31  	 */
32  	public DirChooser(Component parent) {
33  		this.parent = parent;
34  		dirChooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
35  		dirChooser.setDialogTitle(String.format(I18n.INSTANCE.get("dirchooser.title"),
36  			Configuration.INSTANCE.get("application.name")));
37  		dirChooser.setApproveButtonText(I18n.INSTANCE.get("dirchooser.approve.btn"));
38  	}
39  
40  	/**
41  	 * Opens the folder selection.
42  	 * <p>
43  	 * This method will be called by an {@code Observable} element:
44  	 * <ul>
45  	 *   <li>The {@link MenuBar};</li>
46  	 *   <li>Or the {@link SaveMsgField}.</li>
47  	 * </ul>
48  	 * </p>
49  	 *
50  	 * @param o the observable element which will notify this class.
51  	 * @param arg optional parameters (not used).
52  	 */
53  	@Override
54  	public void update(Observable o, Object arg) {
55  		if (o instanceof MenuBar || o instanceof SaveMsgField) {
56  			openFolderSelection();
57  		}
58  	}
59  
60  	/**
61  	 * Opens the folder selection dialog and notify observers once the directory is selected.
62  	 * <p>
63  	 * The only observer notified is the {@link SaveMsgField}.
64  	 * </p>
65  	 */
66  	private void openFolderSelection() {
67  		File filePath = new File(Configuration.INSTANCE.get("emails.default.dir"));
68  		dirChooser.setCurrentDirectory(filePath);
69  
70  		int result = dirChooser.showOpenDialog(parent);
71  
72  		if (result == JFileChooser.APPROVE_OPTION) {
73  			File selectedDir = dirChooser.getSelectedFile();
74  			if (selectedDir != null) {
75  				UIModel.INSTANCE.setSavePath(selectedDir.getAbsolutePath());
76  				setChanged();
77  				notifyObservers();
78  			}
79  		}
80  	}
