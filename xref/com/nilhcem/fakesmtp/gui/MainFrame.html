1   package com.nilhcem.fakesmtp.gui;
2   
3   import com.nilhcem.fakesmtp.core.Configuration;
4   import com.nilhcem.fakesmtp.core.exception.UncaughtExceptionHandler;
5   import com.nilhcem.fakesmtp.server.SMTPServerHandler;
6   import org.slf4j.LoggerFactory;
7   
8   import javax.swing.JFrame;
9   import java.awt.Dimension;
10  import java.awt.Toolkit;
11  import java.awt.event.WindowAdapter;
12  import java.awt.event.WindowEvent;
13  import java.io.IOException;
14  
15  /**
16   * Provides the main window of the application.
17   *
18   * @author Nilhcem
19   * @since 1.0
20   */
21  public final class MainFrame extends WindowAdapter {
22  
23  	private final JFrame mainFrame = new JFrame(Configuration.INSTANCE.get("application.title"));
24  	private final MenuBar menu = new MenuBar();
25  	private final MainPanel panel = new MainPanel(menu);
26  
27  	/**
28  	 * Creates the main window and makes it visible.
29  	 * <p>
30  	 * First, assigns the main panel to the default uncaught exception handler to display exceptions in this panel.<br><br>
31  	 * Before creating the main window, the application will have to set some elements, such as:
32       * </p>
33  	 * <ul>
34  	 *   <li>The minimum and default size;</li>
35  	 *   <li>The menu bar and the main panel;</li>
36  	 *   <li>An icon image;</li>
37  	 *   <li>A shutdown hook to stop the server, once the main window is closed.</li>
38  	 * </ul><br>
39       * <p>
40  	 * The icon of the application is a modified version from the one provided in "{@code WebAppers.com}"
41  	 * <i>(Creative Commons Attribution 3.0 License)</i>.
42       * </p>
43  	 */
44  	public MainFrame() {
45  		((UncaughtExceptionHandler) Thread.getDefaultUncaughtExceptionHandler()).setParentComponent(panel.get());
46  		Dimension frameSize = new Dimension(Integer.parseInt(Configuration.INSTANCE.get("application.min.width")),
47  			Integer.parseInt(Configuration.INSTANCE.get("application.min.height")));
48  
49  		mainFrame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
50  		mainFrame.addWindowListener(this); // for catching windowClosing event
51  		mainFrame.setSize(frameSize);
52  		mainFrame.setMinimumSize(frameSize);
53  
54  		mainFrame.setJMenuBar(menu.get());
55  		mainFrame.getContentPane().add(panel.get());
56  		mainFrame.setLocationRelativeTo(null); // Center main frame
57  		mainFrame.setIconImage(Toolkit.getDefaultToolkit().getImage(getClass().
58  			getResource(Configuration.INSTANCE.get("application.icon.path"))));
59  
60  		// Add shutdown hook to stop server if enabled
61  		Runtime.getRuntime().addShutdownHook(new Thread() {
62  			public void run() {
63  				SMTPServerHandler.INSTANCE.stopServer();
64  			};
65  		});
66  
67  		// Restore last saved smtp port and emails directory
68  		String smtpPort = Configuration.INSTANCE.get("smtp.default.port");
69  		if (smtpPort != null && !smtpPort.isEmpty()) {
70  			panel.getPortText().get().setText(smtpPort);
71  		}
72  		String emailsDir = Configuration.INSTANCE.get("emails.default.dir");
73  		if (emailsDir != null && !emailsDir.isEmpty()) {
74  			panel.getSaveMsgTextField().get().setText(emailsDir);
75  		}
76  
77  		mainFrame.setVisible(true);
78  	}
79  
80  	@Override
81  	public void windowClosing(WindowEvent e) {
82  		// Save configuration
83  		Configuration.INSTANCE.set("smtp.default.port", panel.getPortText().get().getText());
84  		Configuration.INSTANCE.set("emails.default.dir", panel.getSaveMsgTextField().get().getText());
85  		try {
86  			Configuration.INSTANCE.saveToUserProfile();
87  		} catch (IOException ex) {
88  			LoggerFactory.getLogger(MainFrame.class).error("Could not save configuration", ex);
89  		}
90  		// Check for SMTP server running and stop it
91  		if (SMTPServerHandler.INSTANCE.getSmtpServer().isRunning()) {
92  			SMTPServerHandler.INSTANCE.getSmtpServer().stop();
93  		}
94  		mainFrame.dispose();
95  	}
96  }
