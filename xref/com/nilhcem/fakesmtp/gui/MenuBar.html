107 	 * Creates the help menu.

109 	 * The help menu contains an "About" item, to display some software information.
110 	 * </p>
111 	 *
112 	 * @return the newly created help menu.
113 	 */
114 	private JMenu createHelpMenu() {
115 		JMenu helpMenu = new JMenu(i18n.get("menubar.help"));
116 		helpMenu.setMnemonic(i18n.get("menubar.mnemo.help").charAt(0));
117 
118 		JMenuItem about = new JMenuItem(i18n.get("menubar.about"));
119 		about.setMnemonic(i18n.get("menubar.mnemo.about").charAt(0));
120 		about.addActionListener(new ActionListener() {
121 			@Override
122 			public void actionPerformed(ActionEvent e) {
123 				// for copying style
124 				JLabel label = new JLabel();
125 				Font font = label.getFont();
126 
127 				// create some css from the label's font
128 				StringBuffer style = new StringBuffer("font-family:")
129 					.append(font.getFamily()).append(";font-weight:");
130 				if (font.isBold()) {
131 					style.append("bold");
132 				} else {
133 					style.append("normal");
134 				}
135 				style.append(";font-size:").append(font.getSize()).append("pt;");
136 
137 				// html content
138 				String link = i18n.get("menubar.about.dialog.link");
139 				JEditorPane ep = new JEditorPane("text/html",
140 						String.format("<html><body style=\"%s\">%s<br /><a href=\"%s\">%s</a></body></html>",
141 								style, i18n.get("menubar.about.dialog"), link, link));
142 
143 				// handle link events
144 				ep.addHyperlinkListener(new HyperlinkListener() {
145 					@Override
146 					public void hyperlinkUpdate(HyperlinkEvent e) {
147 						if (e.getEventType().equals(HyperlinkEvent.EventType.ACTIVATED)) {
148 							MenuBar.launchUrl(e.getURL().toString());
149 						}
150 					}
151 				});
152 				ep.setEditable(false);
153 				ep.setBackground(label.getBackground());
154 
155 				// show
156 				JOptionPane.showMessageDialog(menuBar.getParent(), ep, String.format(i18n.get("menubar.about.title"),
157 						Configuration.INSTANCE.get("application.name")), JOptionPane.INFORMATION_MESSAGE);
158 			}
159 		});
160 
161 		helpMenu.add(about);
162 		return helpMenu;
163 	}

165 	private static void launchUrl(String url) {
166 		if (Desktop.isDesktopSupported()) {
167 			try {
168 				Desktop desktop = Desktop.getDesktop();
169 				if (desktop != null && desktop.isSupported(Desktop.Action.BROWSE)) {
170 					desktop.browse(new URI(url));
171 				}
172 			} catch (Exception e) {
173 				LOGGER.error("", e);
174 			}
175 		}
176 	}
