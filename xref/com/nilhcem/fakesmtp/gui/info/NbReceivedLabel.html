1   package com.nilhcem.fakesmtp.gui.info;
2   
3   import java.awt.Font;
4   import java.util.Observable;
5   import java.util.Observer;
6   
7   import javax.swing.JLabel;
8   
9   import com.nilhcem.fakesmtp.model.UIModel;
10  import com.nilhcem.fakesmtp.server.MailSaver;
11  
12  /**
13   * Label class to display the number of received emails.
14   *
15   * @author Nilhcem
16   * @since 1.0
17   */
18  public final class NbReceivedLabel implements Observer {
19  	private final JLabel nbReceived = new JLabel("0");
20  
21  	/**
22  	 * Creates the label class (with a bold font).
23  	 */
24  	public NbReceivedLabel() {
25  		Font boldFont = new Font(nbReceived.getFont().getName(), Font.BOLD, nbReceived.getFont().getSize());
26  		nbReceived.setFont(boldFont);
27  	}
28  
29  	/**
30  	 * Returns the JLabel object.
31  	 *
32  	 * @return the JLabel object.
33  	 */
34  	public JLabel get() {
35  		return nbReceived;
36  	}
37  
38  	/**
39  	 * Actions which will be done when the component will be notified by an Observable object.
40  	 * <ul>
41  	 *   <li>If the observable element is a {@link MailSaver} object, the method will increment
42  	 *   the number of received messages and update the {@link UIModel};</li>
43  	 *   <li>If the observable element is a {@link ClearAllButton}, the method will reinitialize
44  	 *   the number of received messages and update the {@link UIModel}.</li>
45  	 * </ul>
46  	 *
47  	 * @param o the observable element which will notify this class.
48  	 * @param arg optional parameters (not used).
49  	 */
50  	@Override
51  	public void update(Observable o, Object arg) {
52  		if (o instanceof MailSaver) {
53  			UIModel model = UIModel.INSTANCE;
54  			int countMsg = model.getNbMessageReceived() + 1;
55  			model.setNbMessageReceived(countMsg);
56  			nbReceived.setText(Integer.toString(countMsg));
57  		} else if (o instanceof ClearAllButton) {
58  			UIModel.INSTANCE.setNbMessageReceived(0);
59  			nbReceived.setText("0");
60  		}
61  	}
62  }
